QSYS/RUNSQL SQL('
  SELECT 
    USER_NAME,
    STATUS,
    SPECIAL_AUTHORITIES 
  FROM QSYS2.USER_INFO 
  WHERE USER_NAME = ''PENTEST01''
') COMMIT(*NONE) OUTPUT(*PRINT)

QSYS/RUNSQL SQL('
  SELECT GROUP_PROFILE_NAME 
  FROM QSYS2.GROUP_PROFILE_ENTRIES 
  WHERE USER_PROFILE_NAME = ''PENTEST01''
') COMMIT(*NONE) OUTPUT(*PRINT)
